Schema

User
  • user_ID(primary key) --- INTEGER(AUTO INCREMENT)
  • username --- TEXT
  • email --- TEXT
  • password --- TEXT
Board
  • board_ID(primary key) --- INTEGER(AUTO INCREMENT)
  • description --- TEXT
  • user_ID (foreign key to User) --- INTEGER
Item
  • user_ID(primary key) --- INTEGER(AUTO INCREMENT)
  • description --- TEXT
  • status(0 for to-do, 1 for in-progress, 2 for done) --- INTEGER
  • user_ID (foreign to User) --- INTEGER
  • board_ID (foreign to Board) --- INTEGER
  • expected pomodoros --- INTEGER
  • completed --- INTEGER
